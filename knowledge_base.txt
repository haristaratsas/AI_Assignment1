Friendly(x) & Loves(x, y) -> Happy(y)
Generous(x) -> Kind(x)
Helpful(x) -> Friendly(x)
Grateful(y) & Loves(x, y) -> Loves(y, x)
Happy(y) -> Grateful(y)
Generous(x) -> Shares(x, y)
Shares(x, y) & Needs(y) -> Happy(y)
Loves(y, x) -> Grateful(x)
